\end{minipage}\hfill
\begin{minipage}[t]{0.54\linewidth}
\textbf{Why it gets hard as the snake grows}
\begin{itemize}
  \item More body segments $\Rightarrow$ more constraints, less free space
  \item Short-sighted greedy policy closes off its own escape routes
  \item Reward signal becomes very sparse: death is rare at length 3, almost certain at length 20+
  \item Agent must plan \textbf{many steps ahead} to leave open passages
\end{itemize}

\medskip
\textbf{Solutions We Employ}
\begin{itemize}
  \item[\pro] \textbf{Dense reward shaping:} $+\Delta d$ potential toward food each step
  \item[\pro] \textbf{Loop penalty:} $-8$ when a cell is revisited $\geq\!4\!\times$ in a 20-step window
  \item[\pro] \textbf{Timeout penalty:} $-5$ on episode timeout to discourage stalling
  \item[\pro] \textbf{Double DQN:} decouple action selection from target evaluation $\Rightarrow$ reduces Q-value overestimation
  \item[\pro] \textbf{N-step returns:} $G_t = \sum_{k=0}^{n-1}\!\gamma^k r_{t+k}$ for longer credit propagation
  \item[\pro] \textbf{Curriculum:} init\_length $3\!\to\!7\!\to\!15$ before switching to hard env
\end{itemize}

\bigskip
\textbf{Dueling Network Architecture}

\smallskip
\begin{tikzpicture}[
  node distance=0.35cm and 0.6cm,
  box/.style={draw=primary!60, fill=primary!8, rounded corners=3pt,
              minimum width=2.8cm, minimum height=0.62cm, font=\small, align=center},
  hbox/.style={draw=primary, fill=primary!22, rounded corners=3pt,
               minimum width=2.3cm, minimum height=0.55cm, font=\small, align=center}]
  \node[box] (trunk) {Shared Trunk};
  \node[hbox, above right=0.5cm and 0.8cm of trunk] (val) {Value $V(s)$};
  \node[hbox, below right=0.5cm and 0.8cm of trunk] (adv) {Adv.\ $A(s,a)$};
  \node[box,  right=4.8cm of trunk]                 (out) {$Q(s,a)$};
  \draw[->,thick] (trunk.east) -- ++(0.4,0) |- (val.west);
  \draw[->,thick] (trunk.east) -- ++(0.4,0) |- (adv.west);
  \draw[->,thick] (val.east)   -- ++(0.3,0) -| (out.north);
  \draw[->,thick] (adv.east)   -- ++(0.3,0) -| (out.south);
  \node[below=0.08cm of out, font=\scriptsize, align=center]
    {$Q = V + A - \overline{A}$};
\end{tikzpicture}

\smallskip
{\small Separating state value $V(s)$ from action advantage $A(s,a)$ stabilises learning in states where the \textit{choice of action matters little} (e.g.\ open space). The advantage head focuses on relative action quality.}

\end{minipage}\hfill
\begin{minipage}[t]{0.50\linewidth}

\textbf{Full Strategy Overview}

\medskip
\renewcommand{\arraystretch}{1.35}
\begin{tabular}{@{}lp{5.7cm}@{}}
\toprule
\textbf{Technique} & \textbf{Why it helps} \\
\midrule
Frame stacking ($n\!=\!4$) &
  Encodes obstacle velocity; converts a partially-observable state into an approximately Markovian one \\
Dueling DQN &
  Separates $V(s)$ from $A(s,a)$; faster convergence when action choice is irrelevant \\
Double DQN &
  Decouples action selection from target evaluation; less Q-value overestimation \\
$n$-step returns ($n\!=\!3$) &
  $G_t = \sum_{k=0}^{2}\gamma^k r_{t+k}$; propagates rewards further back in time \\
Soft target update ($\tau\!=\!0.005$) &
  Polyak averaging of target weights; smoother training than hard copies \\
Curriculum learning &
  Simple env $\to$ long snake $\to$ hard env; avoids early failure spirals \\
Distance shaping &
  Potential-based reward toward nearest gold/silver each step \\
Loop penalty ($-8$) &
  Penalises revisiting the same cell $\geq\!4\!\times$ in 20 steps; breaks self-looping behaviour \\
\bottomrule
\end{tabular}
